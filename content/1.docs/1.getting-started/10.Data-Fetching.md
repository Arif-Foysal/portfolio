## Fetch API Fundamentals

React Native's built-in fetch API for making HTTP requests.

```jsx path=null start=null
import React, { useState, useEffect } from 'react';
import { View, Text, FlatList, ActivityIndicator, Alert, StyleSheet } from 'react-native';

const BasicFetchExample = () => {
  const [posts, setPosts] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetchPosts();
  }, []);

  const fetchPosts = async () => {
    try {
      setLoading(true);
      setError(null);

      const response = await fetch('https://jsonplaceholder.typicode.com/posts');
      
      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }
      
      const data = await response.json();
      setPosts(data.slice(0, 10)); // Limit to first 10 posts
      
    } catch (err) {
      setError(err.message);
      Alert.alert('Error', 'Failed to fetch posts');
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <View style={styles.centerContainer}>
        <ActivityIndicator size="large" color="#007AFF" />
        <Text style={styles.loadingText}>Loading posts...</Text>
      </View>
    );
  }

  if (error) {
    return (
      <View style={styles.centerContainer}>
        <Text style={styles.errorText}>Error: {error}</Text>
        <Button title="Retry" onPress={fetchPosts} />
      </View>
    );
  }

  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <View style={styles.postItem}>
          <Text style={styles.postTitle}>{item.title}</Text>
          <Text style={styles.postBody} numberOfLines={3}>
            {item.body}
          </Text>
        </View>
      )}
      style={styles.container}
    />
  );
};
```

## Advanced Fetch Patterns

Enhanced fetch with request configuration, headers, and error handling.

```jsx path=null start=null
import React, { useState } from 'react';

// API configuration
const API_BASE_URL = 'https://jsonplaceholder.typicode.com';
const DEFAULT_TIMEOUT = 10000;

// API utility functions
const createApiRequest = (endpoint, options = {}) => {
  const url = `${API_BASE_URL}${endpoint}`;
  
  const defaultOptions = {
    timeout: DEFAULT_TIMEOUT,
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
    },
  };

  const config = {
    ...defaultOptions,
    ...options,
    headers: {
      ...defaultOptions.headers,
      ...options.headers,
    },
  };

  // Add timeout support
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), config.timeout);

  return fetch(url, {
    ...config,
    signal: controller.signal,
  }).finally(() => {
    clearTimeout(timeoutId);
  });
};

// HTTP methods wrapper
const api = {
  get: async (endpoint, options = {}) => {
    const response = await createApiRequest(endpoint, {
      method: 'GET',
      ...options,
    });
    return handleResponse(response);
  },

  post: async (endpoint, data, options = {}) => {
    const response = await createApiRequest(endpoint, {
      method: 'POST',
      body: JSON.stringify(data),
      ...options,
    });
    return handleResponse(response);
  },

  put: async (endpoint, data, options = {}) => {
    const response = await createApiRequest(endpoint, {
      method: 'PUT',
      body: JSON.stringify(data),
      ...options,
    });
    return handleResponse(response);
  },

  delete: async (endpoint, options = {}) => {
    const response = await createApiRequest(endpoint, {
      method: 'DELETE',
      ...options,
    });
    return handleResponse(response);
  },
};

// Response handler
const handleResponse = async (response) => {
  if (!response.ok) {
    const errorData = await response.json().catch(() => ({}));
    throw new Error(
      errorData.message || 
      `HTTP ${response.status}: ${response.statusText}`
    );
  }

  const contentType = response.headers.get('content-type');
  if (contentType && contentType.includes('application/json')) {
    return response.json();
  }
  
  return response.text();
};

// Usage example with CRUD operations
const PostManager = () => {
  const [posts, setPosts] = useState([]);
  const [loading, setLoading] = useState(false);
  const [selectedPost, setSelectedPost] = useState(null);

  // GET - Fetch all posts
  const fetchPosts = async () => {
    try {
      setLoading(true);
      const data = await api.get('/posts?_limit=10');
      setPosts(data);
    } catch (error) {
      Alert.alert('Error', `Failed to fetch posts: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  // POST - Create new post
  const createPost = async (postData) => {
    try {
      setLoading(true);
      const newPost = await api.post('/posts', postData);
      setPosts(prev => [newPost, ...prev]);
      Alert.alert('Success', 'Post created successfully');
    } catch (error) {
      Alert.alert('Error', `Failed to create post: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  // PUT - Update post
  const updatePost = async (postId, updateData) => {
    try {
      setLoading(true);
      const updatedPost = await api.put(`/posts/${postId}`, updateData);
      setPosts(prev => 
        prev.map(post => post.id === postId ? updatedPost : post)
      );
      Alert.alert('Success', 'Post updated successfully');
    } catch (error) {
      Alert.alert('Error', `Failed to update post: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  // DELETE - Delete post
  const deletePost = async (postId) => {
    Alert.alert(
      'Confirm Delete',
      'Are you sure you want to delete this post?',
      [
        { text: 'Cancel', style: 'cancel' },
        {
          text: 'Delete',
          style: 'destructive',
          onPress: async () => {
            try {
              setLoading(true);
              await api.delete(`/posts/${postId}`);
              setPosts(prev => prev.filter(post => post.id !== postId));
              Alert.alert('Success', 'Post deleted successfully');
            } catch (error) {
              Alert.alert('Error', `Failed to delete post: ${error.message}`);
            } finally {
              setLoading(false);
            }
          },
        },
      ]
    );
  };

  useEffect(() => {
    fetchPosts();
  }, []);

  return (
    <View style={styles.container}>
      {/* Your UI components here */}
    </View>
  );
};
```

## Custom Hooks for Data Fetching

Reusable hooks for common data fetching patterns.

```jsx path=null start=null
import { useState, useEffect, useCallback, useRef } from 'react';

// Generic data fetching hook
export const useApi = (apiFunction, dependencies = [], immediate = true) => {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(immediate);
  const [error, setError] = useState(null);
  const cancelRef = useRef(false);

  const execute = useCallback(async (...args) => {
    try {
      cancelRef.current = false;
      setLoading(true);
      setError(null);

      const result = await apiFunction(...args);
      
      if (!cancelRef.current) {
        setData(result);
      }
    } catch (err) {
      if (!cancelRef.current) {
        setError(err);
      }
    } finally {
      if (!cancelRef.current) {
        setLoading(false);
      }
    }
  }, [apiFunction]);

  useEffect(() => {
    if (immediate) {
      execute();
    }

    return () => {
      cancelRef.current = true;
    };
  }, dependencies);

  const refetch = useCallback(() => execute(), [execute]);

  return { data, loading, error, execute, refetch };
};

// Hook for paginated data
export const usePaginatedApi = (apiFunction, pageSize = 20) => {
  const [data, setData] = useState([]);
  const [loading, setLoading] = useState(false);
  const [loadingMore, setLoadingMore] = useState(false);
  const [error, setError] = useState(null);
  const [hasMore, setHasMore] = useState(true);
  const [page, setPage] = useState(1);

  const loadPage = useCallback(async (pageNumber, reset = false) => {
    try {
      if (pageNumber === 1) {
        setLoading(true);
      } else {
        setLoadingMore(true);
      }
      
      setError(null);

      const result = await apiFunction({
        page: pageNumber,
        limit: pageSize,
      });

      const newItems = result.data || result;
      const hasMoreItems = newItems.length === pageSize;

      setData(prev => reset ? newItems : [...prev, ...newItems]);
      setHasMore(hasMoreItems);
      setPage(pageNumber);

    } catch (err) {
      setError(err);
    } finally {
      setLoading(false);
      setLoadingMore(false);
    }
  }, [apiFunction, pageSize]);

  const loadInitial = useCallback(() => {
    loadPage(1, true);
  }, [loadPage]);

  const loadMore = useCallback(() => {
    if (!loadingMore && hasMore) {
      loadPage(page + 1);
    }
  }, [loadPage, page, loadingMore, hasMore]);

  const refresh = useCallback(() => {
    setData([]);
    setPage(1);
    setHasMore(true);
    loadPage(1, true);
  }, [loadPage]);

  useEffect(() => {
    loadInitial();
  }, []);

  return {
    data,
    loading,
    loadingMore,
    error,
    hasMore,
    loadMore,
    refresh,
    refetch: loadInitial,
  };
};

// Hook for search with debouncing
export const useSearchApi = (searchFunction, debounceMs = 500) => {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const timeoutRef = useRef();

  const search = useCallback(async (searchQuery) => {
    if (!searchQuery.trim()) {
      setResults([]);
      return;
    }

    try {
      setLoading(true);
      setError(null);
      
      const data = await searchFunction(searchQuery);
      setResults(data);
    } catch (err) {
      setError(err);
      setResults([]);
    } finally {
      setLoading(false);
    }
  }, [searchFunction]);

  useEffect(() => {
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }

    if (query) {
      timeoutRef.current = setTimeout(() => {
        search(query);
      }, debounceMs);
    } else {
      setResults([]);
    }

    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, [query, search, debounceMs]);

  return {
    query,
    setQuery,
    results,
    loading,
    error,
  };
};

// Usage examples
const PostsList = () => {
  const fetchPosts = useCallback(
    () => api.get('/posts'),
    []
  );

  const { data: posts, loading, error, refetch } = useApi(fetchPosts, [], true);

  if (loading) return <LoadingSpinner />;
  if (error) return <ErrorMessage error={error} onRetry={refetch} />;

  return (
    <FlatList
      data={posts}
      renderItem={({ item }) => <PostItem post={item} />}
      refreshing={loading}
      onRefresh={refetch}
    />
  );
};

const InfinitePostsList = () => {
  const fetchPostsPage = useCallback(
    ({ page, limit }) => api.get(`/posts?_page=${page}&_limit=${limit}`),
    []
  );

  const {
    data: posts,
    loading,
    loadingMore,
    error,
    hasMore,
    loadMore,
    refresh,
  } = usePaginatedApi(fetchPostsPage, 10);

  return (
    <FlatList
      data={posts}
      renderItem={({ item }) => <PostItem post={item} />}
      onEndReached={loadMore}
      onEndReachedThreshold={0.1}
      ListFooterComponent={() => (
        loadingMore ? <LoadingSpinner /> : null
      )}
      refreshing={loading}
      onRefresh={refresh}
    />
  );
};

const SearchScreen = () => {
  const searchPosts = useCallback(
    (searchTerm) => api.get(`/posts?q=${encodeURIComponent(searchTerm)}`),
    []
  );

  const { query, setQuery, results, loading, error } = useSearchApi(searchPosts, 300);

  return (
    <View style={styles.container}>
      <TextInput
        style={styles.searchInput}
        value={query}
        onChangeText={setQuery}
        placeholder="Search posts..."
      />
      
      {loading && <LoadingSpinner />}
      {error && <ErrorMessage error={error} />}
      
      <FlatList
        data={results}
        renderItem={({ item }) => <PostItem post={item} />}
        keyExtractor={(item) => item.id.toString()}
      />
    </View>
  );
};
```

```jsx path=null start=null
import axios from 'axios';

const apiClient = axios.create({
  baseURL: 'https://jsonplaceholder.typicode.com',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Request interceptor
apiClient.interceptors.request.use(
  (config) => {
    // Add auth token if available
    const token = getAuthToken(); // Your token retrieval logic
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    
    console.log('Request:', config.method?.toUpperCase(), config.url);
    return config;
  },
  (error) => {
    console.error('Request error:', error);
    return Promise.reject(error);
  }
);

// Response interceptor
apiClient.interceptors.response.use(
  (response) => {
    console.log('Response:', response.status, response.config.url);
    return response;
  },
  async (error) => {
    const originalRequest = error.config;

    // Handle 401 unauthorized
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true;
      
      try {
        const newToken = await refreshAuthToken(); // Your token refresh logic
        originalRequest.headers.Authorization = `Bearer ${newToken}`;
        return apiClient(originalRequest);
      } catch (refreshError) {
        // Redirect to login
        handleLogout();
        return Promise.reject(refreshError);
      }
    }

    console.error('Response error:', error.response?.status, error.message);
    return Promise.reject(error);
  }
);

// API service functions
export const postService = {
  getAll: (params = {}) =>

    apiClient.get(`/posts/${id}`).then(response => response.data),

  create: (data) => 
    apiClient.post('/posts', data).then(response => response.data),

    apiClient.put(`/posts/${id}`, data).then(response => response.data),

  delete: (id) => 
    apiClient.delete(`/posts/${id}`).then(response => response.data),

  search: (query) =>
    apiClient.get('/posts', {

const PostsWithAxios = () => {
```

## Network State & Offline Handling

Handle network connectivity and offline scenarios.

```jsx path=null start=null
import NetInfo from '@react-native-community/netinfo';
import AsyncStorage from '@react-native-async-storage/async-storage';

// Network status hook
export const useNetworkStatus = () => {
  const [isConnected, setIsConnected] = useState(true);
  const [connectionType, setConnectionType] = useState('unknown');

  useEffect(() => {
    const unsubscribe = NetInfo.addEventListener(state => {
      setIsConnected(state.isConnected ?? false);
      setConnectionType(state.type);
    });

    return unsubscribe;
  }, []);

  return { isConnected, connectionType };
};

// Offline-aware data fetching
export const useOfflineData = (key, fetchFunction, dependencies = []) => {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [isFromCache, setIsFromCache] = useState(false);
  const { isConnected } = useNetworkStatus();

  const loadCachedData = async () => {
    try {
      const cached = await AsyncStorage.getItem(`cache_${key}`);
      if (cached) {
        const { data: cachedData, timestamp } = JSON.parse(cached);
        const isExpired = Date.now() - timestamp > 5 * 60 * 1000; // 5 minutes
        
        if (!isExpired) {
          setData(cachedData);
          setIsFromCache(true);
          return true;
        }
      }
    } catch (err) {
      console.error('Failed to load cached data:', err);
    }
    return false;
  };

  const saveToCache = async (data) => {
    try {
      await AsyncStorage.setItem(`cache_${key}`, JSON.stringify({
        data,
        timestamp: Date.now(),
      }));
    } catch (err) {
      console.error('Failed to save to cache:', err);
    }
  };

  const fetchData = async () => {
    if (!isConnected) {
      const hasCached = await loadCachedData();
      if (!hasCached) {
        setError('No internet connection and no cached data');
      }
      setLoading(false);
      return;
    }

    try {
      setLoading(true);
      setError(null);
      
      const result = await fetchFunction();
      setData(result);
      setIsFromCache(false);
      await saveToCache(result);
      
    } catch (err) {
      setError(err.message);
      
      // Try to load cached data as fallback
      const hasCached = await loadCachedData();
      if (hasCached) {
        setError('Using cached data due to network error');
      }
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
  }, [isConnected, ...dependencies]);

  const refetch = useCallback(() => {
    if (isConnected) {
      fetchData();
    }
  }, [isConnected, fetchData]);

  return { 
    data, 
    loading, 
    error, 
    isFromCache, 
    isConnected, 
    refetch 
  };
};

// Usage example
const OfflinePostsList = () => {
  const fetchPosts = useCallback(
    () => api.get('/posts?_limit=10'),
    []
  );

  const { 
    data: posts, 
    loading, 
    error, 
    isFromCache, 
    isConnected,
    refetch 
  } = useOfflineData('posts', fetchPosts, []);

  return (
    <View style={styles.container}>
      {!isConnected && (
        <View style={styles.offlineBar}>
          <Text style={styles.offlineText}>No internet connection</Text>
        </View>
      )}
      
      {isFromCache && (
        <View style={styles.cacheBar}>
          <Text style={styles.cacheText}>Showing cached data</Text>
          {isConnected && (
            <TouchableOpacity onPress={refetch}>
              <Text style={styles.refreshText}>Refresh</Text>
            </TouchableOpacity>
          )}
        </View>
      )}

      {loading ? (
        <LoadingSpinner />
      ) : error && !posts ? (
        <ErrorMessage error={error} onRetry={refetch} />
      ) : (
        <FlatList
          data={posts}
          renderItem={({ item }) => <PostItem post={item} />}
          refreshing={loading}
          onRefresh={refetch}
        />
      )}
    </View>
  );
};
```

## 💡 Key Takeaways

1. **Always handle loading and error states** in data fetching
2. **Use proper HTTP status code checking** with fetch
3. **Implement request timeouts** to prevent hanging requests
4. **Custom hooks** make data fetching logic reusable
5. **Axios interceptors** provide powerful request/response processing
6. **Cache data locally** for offline functionality
7. **Monitor network status** for better user experience
8. **Implement retry logic** for failed requests
