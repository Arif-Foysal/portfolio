## Advanced Image Handling

Loading states, error handling, and performance optimizations.

```jsx path=null start=null
import React, { useState } from 'react';
import {
  View,
  Image,
  Text,
  ActivityIndicator,
  StyleSheet,
  TouchableOpacity,
} from 'react-native';

const AdvancedImage = ({ source, style, ...props }) => {
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(false);
  const [imageLoaded, setImageLoaded] = useState(false);

  const handleLoadStart = () => {
    setLoading(true);
    setError(false);
    setImageLoaded(false);
  };

  const handleLoad = () => {
    setLoading(false);
    setImageLoaded(true);
    setError(false);
  };

  const handleError = () => {
    setLoading(false);
    setError(true);
    setImageLoaded(false);
  };

  const retry = () => {
    setError(false);
    setLoading(true);
    // Force re-render by creating new source object
    setImageLoaded(false);
  };

  return (
    <View style={[styles.imageContainer, style]}>
      <Image
        {...props}
        source={source}
        style={[styles.image, style]}
        onLoadStart={handleLoadStart}
        onLoad={handleLoad}
        onError={handleError}
        // Add cache control for remote images
        {...(typeof source === 'object' && source.uri && {
          source: {
            ...source,
            cache: 'force-cache', // or 'default', 'reload', 'no-cache'
          }
        })}
      />

      {/* Loading indicator */}
      {loading && !error && (
        <View style={styles.loadingOverlay}>
          <ActivityIndicator size="small" color="#007AFF" />
        </View>
      )}

      {/* Error state */}
      {error && (
        <TouchableOpacity style={styles.errorOverlay} onPress={retry}>
          <Text style={styles.errorText}>‚ö†Ô∏è</Text>
          <Text style={styles.retryText}>Tap to retry</Text>

      )}
    </View>
  );
};
```

```jsx path=null start=null
      priority: FastImage.priority.normal,
    }))

// Custom hook for image optimization
import { useState, useEffect } from 'react';
import { Dimensions } from 'react-native';

const useOptimizedImageSize = (originalWidth, originalHeight, containerWidth) => {
  const [optimizedSize, setOptimizedSize] = useState({
    width: containerWidth,
    height: containerWidth,
  });

  useEffect(() => {
    if (originalWidth && originalHeight && containerWidth) {
      const aspectRatio = originalHeight / originalWidth;
      const optimizedHeight = containerWidth * aspectRatio;
      
      setOptimizedSize({
        width: containerWidth,
        height: optimizedHeight,
      });
    }
  }, [originalWidth, originalHeight, containerWidth]);

  return optimizedSize;
};

// Progressive image loading
const ProgressiveImage = ({ thumbnailUri, imageUri, style, ...props }) => {
  const [imageLoaded, setImageLoaded] = useState(false);
  const [thumbnailLoaded, setThumbnailLoaded] = useState(false);

  return (
    <View style={style}>
      {/* Low quality thumbnail */}
      <Image
        source={{ uri: thumbnailUri }}
        style={[styles.imageBackground, { opacity: thumbnailLoaded ? 1 : 0 }]}
        onLoad={() => setThumbnailLoaded(true)}
        blurRadius={1} // Slight blur for thumbnail
      />
      
      {/* High quality image */}
      <Image
        source={{ uri: imageUri }}
        style={[styles.imageBackground, { opacity: imageLoaded ? 1 : 0 }]}
        onLoad={() => setImageLoaded(true)}
        {...props}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  imageBackground: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
  },
});
```

```jsx path=null start=null
import {

import { request, PERMISSIONS, RESULTS } from 'react-native-permissions';

const ImagePickerComponent = () => {
  const [selectedImage, setSelectedImage] = useState(null);
  const [loading, setLoading] = useState(false);

  const requestCameraPermission = async () => {
    const permission = Platform.OS === 'ios' 
      ? PERMISSIONS.IOS.CAMERA 
      : PERMISSIONS.ANDROID.CAMERA;

    const result = await request(permission);
    return result === RESULTS.GRANTED;
  };

  const requestGalleryPermission = async () => {
    const permission = Platform.OS === 'ios'
      ? PERMISSIONS.IOS.PHOTO_LIBRARY
      : PERMISSIONS.ANDROID.READ_EXTERNAL_STORAGE;

    const result = await request(permission);
    return result === RESULTS.GRANTED;
  };

  const showImagePicker = () => {
    const options = {
      title: 'Select Image',
      cancelButtonTitle: 'Cancel',
      takePhotoButtonTitle: 'Take Photo',
      chooseFromLibraryButtonTitle: 'Choose from Gallery',
      storageOptions: {
        skipBackup: true,
        path: 'images',
      },
      mediaType: 'photo',
      maxWidth: 1000,
      maxHeight: 1000,
      quality: 0.8,
    };

    // For newer versions, use Alert to show options
    Alert.alert(
      'Select Image',
      'Choose an option',
      [
        { text: 'Camera', onPress: openCamera },
        { text: 'Gallery', onPress: openGallery },
        { text: 'Cancel', style: 'cancel' },
      ]
    );
  };

  const openCamera = async () => {
    const hasPermission = await requestCameraPermission();
    if (!hasPermission) {
      Alert.alert('Permission denied', 'Camera permission is required');
      return;
    }

    const options = {
      mediaType: 'photo',
      includeBase64: false,
      maxHeight: 1000,
      maxWidth: 1000,
      quality: 0.8,
    };

    setLoading(true);
    launchCamera(options, (response) => {
      setLoading(false);
      if (response.assets && response.assets[0]) {
        setSelectedImage(response.assets[0]);
      }
    });
  };

  const openGallery = async () => {
    const hasPermission = await requestGalleryPermission();
    if (!hasPermission) {
      Alert.alert('Permission denied', 'Gallery permission is required');
      return;
    }

    const options = {
      mediaType: 'photo',
      includeBase64: false,
      maxHeight: 1000,
      maxWidth: 1000,
      quality: 0.8,
      selectionLimit: 1, // Allow single selection
    };

    setLoading(true);
    launchImageLibrary(options, (response) => {
      setLoading(false);
      if (response.assets && response.assets[0]) {
        setSelectedImage(response.assets[0]);
      }
    });
  };

  const removeImage = () => {
    setSelectedImage(null);
  };

  return (
    <View style={styles.container}>
      <View style={styles.imageContainer}>
        {selectedImage ? (
          <View style={styles.selectedImageContainer}>
            <Image
              source={{ uri: selectedImage.uri }}
              style={styles.selectedImage}
              resizeMode="cover"
            />
            <TouchableOpacity style={styles.removeButton} onPress={removeImage}>
              <Text style={styles.removeButtonText}>√ó</Text>
            </TouchableOpacity>
          </View>
        ) : (
          <TouchableOpacity
            style={styles.placeholderContainer}
            onPress={showImagePicker}
            disabled={loading}
          >
            {loading ? (
              <ActivityIndicator size="large" color="#007AFF" />
            ) : (
              <>
                <Text style={styles.placeholderIcon}>üì∑</Text>
                <Text style={styles.placeholderText}>Tap to add image</Text>
              </>
            )}
          </TouchableOpacity>
        )}
      </View>

      {selectedImage && (
        <View style={styles.imageInfo}>
          <Text style={styles.infoText}>
            Size: {Math.round(selectedImage.fileSize / 1024)} KB
          </Text>
          <Text style={styles.infoText}>
            Dimensions: {selectedImage.width} x {selectedImage.height}
          </Text>
        </View>
      )}

      <TouchableOpacity
        style={[styles.button, styles.primaryButton]}
        onPress={showImagePicker}
        disabled={loading}
      >
```

```jsx path=null start=null
// Using react-native-video
import Video from 'react-native-video';
import { useState, useRef } from 'react';

const VideoPlayer = ({ source, style }) => {
  const [paused, setPaused] = useState(true);
  const [currentTime, setCurrentTime] = useState(0);
  const [duration, setDuration] = useState(0);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const videoRef = useRef();

  const onLoad = (data) => {
    setDuration(data.duration);
    setLoading(false);
  };

  const onProgress = (data) => {
    setCurrentTime(data.currentTime);
  };

  const onError = (error) => {
    setError(error);
    setLoading(false);
  };

  const togglePlayPause = () => {
    setPaused(!paused);
  };

  const seekTo = (time) => {
    videoRef.current.seek(time);
  };

  const formatTime = (timeInSeconds) => {
    const minutes = Math.floor(timeInSeconds / 60);
    const seconds = Math.floor(timeInSeconds % 60);
    return `${minutes}:${seconds.toString().padStart(2, '0')}`;
  };

  if (error) {
    return (
      <View style={[styles.videoContainer, style]}>
        <Text style={styles.errorText}>Failed to load video</Text>
      </View>
    );
  }

  return (
    <View style={[styles.videoContainer, style]}>
      <Video
        ref={videoRef}
        source={source}
        style={styles.video}
        paused={paused}
        resizeMode="contain"
        onLoad={onLoad}
        onProgress={onProgress}
        onError={onError}
        onEnd={() => setPaused(true)}
      />

      {/* Loading overlay */}
      {loading && (
        <View style={styles.loadingOverlay}>
          <ActivityIndicator size="large" color="#fff" />
        </View>
      )}

      {/* Controls */}
      <View style={styles.controlsContainer}>
        <TouchableOpacity style={styles.playButton} onPress={togglePlayPause}>
          <Text style={styles.playButtonText}>
            {paused ? '‚ñ∂Ô∏è' : '‚è∏Ô∏è'}
          </Text>
        </TouchableOpacity>

        <View style={styles.timeContainer}>
          <Text style={styles.timeText}>
            {formatTime(currentTime)} / {formatTime(duration)}

        </View>
      </View>

        <View style={styles.progressBackground}>
          <View
            style={[
              styles.progressFill,
              {
                width: duration > 0 ? `${(currentTime / duration) * 100}%` : '0%',
              },
            ]}
          />
        </View>
      </View>
    </View>
  );
};
```

### Dynamic Asset Loading

```jsx path=null start=null
// For dynamic image loading based on themes or conditions
const ThemeAwareImage = ({ imageName, theme = 'light', ...props }) => {
  const getThemedImage = (name, currentTheme) => {
    const themeMap = {
      light: {
        logo: require('../assets/images/logo-light.png'),
        background: require('../assets/images/bg-light.png'),
      },
      dark: {
        logo: require('../assets/images/logo-dark.png'),
        background: require('../assets/images/bg-dark.png'),
      },
    };

    return themeMap[currentTheme]?.[name] || themeMap.light[name];
  };

  return (
    <Image
      source={getThemedImage(imageName, theme)}
      {...props}
    />
  );
};

// Responsive image component
const ResponsiveImage = ({ source, maxWidth, aspectRatio = 1, ...props }) => {
  const { width: screenWidth } = Dimensions.get('window');
  
  const imageWidth = Math.min(maxWidth || screenWidth, screenWidth - 40);
  const imageHeight = imageWidth / aspectRatio;

  return (
    <Image
      source={source}
      style={{
        width: imageWidth,
        height: imageHeight,
      }}
      {...props}
    />
  );
};
```
